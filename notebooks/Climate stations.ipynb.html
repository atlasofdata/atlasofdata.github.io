def get_buckets(grouped, buckets_nb=40):    
    buckets = np.linspace( grouped.index.min(), grouped.index.max(), buckets_nb)
    buckets_name = [i for i, b in enumerate(buckets) if i < len(buckets)-1]
    labels = ["%s-%s"%(int(b),int(buckets[i+1])) for i, b in enumerate(buckets[:-1])]
    return buckets, buckets_name, labels

# see when they were started
def make_bar_chart(grouped, title):
    # calculate buckets
    buckets, buckets_name, labels = get_buckets(grouped)
    grouped["period"] = pd.cut(grouped.index, buckets, labels=buckets_name )
    


    ax = grouped.groupby(["period"]).sum()["Station ID"].plot(kind='bar', figsize=(20,10))
    ax.set_xticklabels(labels)
    ax.set_title(title, fontsize=15)
    ax.annotate("Source: Berkeley Earth", xy=(0.1, .001), xycoords="figure fraction", color="#555555")
    return ax
